class DoubleEndedPriorityQueue {
  private:
    priority_queue<int> maxHeap;
    priority_queue<int, vector<int>, greater<int>> minHeap;
    unordered_map<int, int> count; // ì¶œí˜„ íšŸìˆ˜

    template <typename T> void clean(T &heap) { // ê°’ ë™ê¸°í™”
        while (!heap.empty() && count[heap.top()] == 0) {
            heap.pop();
        }
    }

  public:
    void insert(int value) {
        maxHeap.push(value);
        minHeap.push(value);
        count[value]++;
    }

    void deleteMax() {
        clean(maxHeap);
        if (!maxHeap.empty()) { // ë¬¸ì œ ë³¸ë¬¸: ë¹„ì–´ìˆë‹¤ë©´ ë¬´ì‹œí•´ë„ ì¢‹ë‹¤.
            int value = maxHeap.top();

        }
    }

    void deleteMin() {
        clean(minHeap);
        if (!minHeap.empty()) {
            int value = minHeap.top();
            minHeap.pop();
            count[value]--;
        }
    }

    int getMax() {
        clean(maxHeap);
        // if (!maxHeap.empty())
        return maxHeap.top();
    }

    int getMin() {
        clean(minHeap);
        // if (!minHeap.empty())
        return minHeap.top();
    }

    bool empty() {
        clean(minHeap);
        return minHeap.empty();
    }
};
